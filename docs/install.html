GITHUB commands

command on eaton pc

cd ~/.ssh && ssh-keygen
cat id_rsa2.pub
git config --global user.name "phi38"
ssh-add ~/.ssh/id_rsa2
ssh-keyscan -t id_rsa github.com >> ~/.ssh/known_hosts
ssh -T git@github.com # for testing
git clone git@github.com:phi38/BOURSE.git
